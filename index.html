            filter: grayscale(1);

            opacity: 0.7;

        }

    </style>

<body class="bg-gray-50 text-gray-800 font-sans">


    <!-- Header -->

    <header class="bg-white shadow-sm py-6 mb-8">

        <div class="max-w-6xl mx-auto px-4 flex justify-between items-center">

            <h1 class="text-2xl font-bold text-gray-900">Lista de Deseos</h1>

            <button id="adminBtn" onclick="toggleAdminMode()" class="px-4 py-2 text-sm bg-gray-200 rounded-full hover:bg-gray-300 transition">

                Modo Admin: OFF

            </button>

        </div>

    </header>


    <!-- Main Content -->

    <main class="max-w-6xl mx-auto px-4 mb-12">

        <div id="loading" class="text-center text-gray-500 text-lg">Cargando elementos...</div>

        <div id="cardsContainer" class="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-6 hidden">

            <!-- Las cards se inyectarán aquí mediante JavaScript -->

        </div>

    </main>


    <script>

        // REPLAZA CON LA URL DE TU SCRIPT DE GOOGLE (Paso 3)

        const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbyIXj_HKVA3vPdx0HZFLfXWzGFGtSoft5f8DDBdCC9rN1WP5H-XsIDjW9_Sjexuck-B/exec"; 

        

        let itemsData = [];

        let isAdmin = false;


        function toggleAdminMode() {

            isAdmin = !isAdmin;

            document.getElementById('adminBtn').innerText = `Modo Admin: ${isAdmin ? 'ON' : 'OFF'}`;

            document.getElementById('adminBtn').classList.toggle('bg-green-200', isAdmin);

            renderCards();

        }


        async function fetchData() {

            try {

                const response = await fetch(`${WEB_APP_URL}?action=read`);

                itemsData = await response.json();

                document.getElementById('loading').classList.add('hidden');

                document.getElementById('cardsContainer').classList.remove('hidden');

                renderCards();

            } catch (error) {

                console.error("Error al cargar los datos:", error);

                document.getElementById('loading').innerText = "Error al cargar los datos. Verifica la configuración.";

            }

        }


        function renderCards() {

            const container = document.getElementById('cardsContainer');

            container.innerHTML = '';


            // Ordenar: Primero los NO comprados, luego los COMPRADOS (enviados al fondo)

            const sortedItems = [...itemsData].sort((a, b) => a.comprado - b.comprado);


            sortedItems.forEach(item => {

                const isComprado = item.comprado === true || item.comprado === "TRUE";

                

                const card = document.createElement('div');

                card.className = `bg-white rounded-2xl shadow-md overflow-hidden flex flex-col justify-between transition transform hover:-translate-y-1 ${isComprado ? 'card-disabled' : ''}`;

                

                // Si está comprado pero es admin, permitimos interactuar con los botones de control

                if (isComprado && isAdmin) card.classList.add('admin-controls');


                card.innerHTML = `

                    <div class="cursor-pointer" onclick="${isComprado ? '' : `window.open('${item.link}', '_blank')`}">

                        <img src="${item.imagen_url || 'https://via.placeholder.com/300x200'}" alt="${item.titulo}" class="w-full h-48 object-cover">

                        <div class="p-5">

                            <h3 class="text-lg font-semibold text-gray-900 mb-2">${item.titulo}</h3>

                            <p class="text-gray-600 text-sm line-clamp-3">${item.descripcion}</p>

                        </div>

                    </div>

                    

                    <div class="p-5 pt-0 mt-auto">

                        ${isComprado ? `

                            <div class="bg-gray-100 p-3 rounded-xl text-center">

                                <p class="text-sm text-gray-500 font-medium">Comprado por: <span class="text-gray-800 font-bold">${item.comprador}</span></p>

                                ${isAdmin ? `

                                    <button onclick="unlockItem(${item.id})" class="mt-2 text-xs text-red-600 underline font-semibold pointer-events-auto">

                                        Desbloquear (Admin)

                                    </button>

                                ` : ''}

                            </div>

                        ` : `

                            <button onclick="markAsBought(${item.id})" class="w-full py-2.5 bg-neutral-900 text-white font-medium text-sm rounded-xl hover:bg-neutral-800 transition">

                                Marcar como comprado

                            </button>

                        `}

                    </div>

                `;

                container.appendChild(card);

            });

        }


        async function markAsBought(id) {

            const name = prompt("Por favor, ingresa tu nombre para reservarlo:");

            if (!name || name.trim() === "") return;


            // Optimistic UI update (actualiza en pantalla inmediatamente)

            const item = itemsData.find(i => i.id == id);

            if (item) {

                item.comprado = true;

                item.comprador = name;

                renderCards();

            }


            // Enviar a la base de datos (Google Sheets)

            await fetch(`${WEB_APP_URL}`, {

                method: 'POST',

                mode: 'no-cors',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ action: 'update', id: id, comprado: true, comprador: name })

            });

            

            // Recargamos para asegurar sincronía

            setTimeout(fetchData, 1500); 

        }


        async function unlockItem(id) {

            if (!confirm("¿Estás seguro de que quieres desbloquear esta card?")) return;


            const item = itemsData.find(i => i.id == id);

            if (item) {

                item.comprado = false;

                item.comprador = "";

                renderCards();

            }


            await fetch(`${WEB_APP_URL}`, {

                method: 'POST',

                mode: 'no-cors',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ action: 'update', id: id, comprado: false, comprador: "" })

            });


            setTimeout(fetchData, 1500);

        }


        // Inicializar

        fetchData();

    </script>
